        Cheatsheet del DSL
        

          .bpm(120)                      tempo
          .track('drums')                cambia/crea pista
          .volume(0.7) .pan(-0.3)        nivel/estéreo de pista
          .mute() .solo()                silenciar/aislar
          .drum('kick'|'snare'|'hat'|'clap') nueva voz drum
          .synth('sine'|'square'|'sawtooth'|'triangle') nueva voz sintética
          .pattern('x...x...x...x...')   16 steps; x=hit, .=silencio
          .notes('c4 d#4 sol4 do5')      notas absolutas (anglo+latina)
          .scale('Cm pent') .notes("1 b3 5 1'") escala + grados
          .filter(800) .release(0.3)     parámetros del synth
          .every(2)                      suena 1 de cada N ciclos
